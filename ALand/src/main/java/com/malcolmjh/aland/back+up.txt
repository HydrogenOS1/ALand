package com.malcolmjh.aland;

import com.malcolmjh.aland.commands.craftcommand;
import org.bukkit.plugin.java.JavaPlugin;

import java.io.IOException;
import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;

public final class ALand extends JavaPlugin {

    @Override
    public void onEnable() {
        //API Start stuff goes here!
          //Main API
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create("https://weatherapi-com.p.rapidapi.com/astronomy.json?q=London"))
                .header("X-RapidAPI-Key", "c9287726c7msh833b730e10975bfp19f501jsnc33f9dd15db2")
                .header("X-RapidAPI-Host", "weatherapi-com.p.rapidapi.com")
                .method("GET", HttpRequest.BodyPublishers.noBody())
                .build();
        HttpResponse<String> response = null;
        try {
            response = HttpClient.newHttpClient().send(request, HttpResponse.BodyHandlers.ofString());
        } catch (IOException e) {
            throw new RuntimeException(e);
        } catch (InterruptedException e) {
            throw new RuntimeException(e);
        }
        System.out.println(response.body());

        //other start logic
        getCommand("craft").setExecutor(new craftcommand());

        //temp error stuff
        int error = 0;

        if (error == 0)
            System.out.println("No error's, Yay!");
            System.out.println("ALand Started!");
        if (error == 1)
            System.out.println("ERROR: 500. Sorry, The API is down! Try again later!");
        if (error == 2)
            System.out.println("ERROR: 510. Sorry, The Database is down! Try again later!");
        if (error == 3)
            System.out.println("ERROR: 409. Sorry, Either you are not conected to the internet or the API and test API are both down!");
        if (error == 4)
            System.out.println("ERROR: 418. I'm a teapot!");
    }

    @Override
    public void onDisable() {
        System.out.println("ALand has stoped");
    }
}
